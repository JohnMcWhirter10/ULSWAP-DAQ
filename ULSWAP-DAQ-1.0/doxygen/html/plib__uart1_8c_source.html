51void static UART1_ErrorClear( void )
52{
53 UART_ERROR errors = UART_ERROR_NONE;
54 uint8_t dummyData = 0u;
55
56 errors = (UART_ERROR)(U1STA & (_U1STA_OERR_MASK | _U1STA_FERR_MASK | _U1STA_PERR_MASK));
57
58 if(errors != UART_ERROR_NONE)
59 {
60 /* If it's a overrun error then clear it to flush FIFO */
61 if(U1STA & _U1STA_OERR_MASK)
62 {
63 U1STACLR = _U1STA_OERR_MASK;
64 }
65
66 /* Read existing error bytes from FIFO to clear parity and framing error flags */
67 while(U1STA & _U1STA_URXDA_MASK)
68 {
69 dummyData = U1RXREG;
70 }
71
72 }
73
74 // Ignore the warning
75 (void)dummyData;
76}

78void UART1_Initialize( void )
79{
80 /* Set up UxMODE bits */
81 /* STSEL = 0 */
82 /* PDSEL = 0 */
83 /* UEN = 0 */
84
85 U1MODE = 0x0;
86
87 /* Enable UART1 Receiver and Transmitter */
88 U1STASET = (_U1STA_UTXEN_MASK | _U1STA_URXEN_MASK | _U1STA_UTXISEL1_MASK );
89
90 /* BAUD Rate register Setup */
91 U1BRG = 25;
92
93 /* Turn ON UART1 */
94 U1MODESET = _U1MODE_ON_MASK;
95}

97bool UART1_SerialSetup( UART_SERIAL_SETUP *setup, uint32_t srcClkFreq )
98{
99 bool status = false;
100 uint32_t baud;
101 uint32_t status_ctrl;
102 bool brgh = 0;
103 int32_t uxbrg = 0;
104
105 if (setup != NULL)
106 {
107 baud = setup->baudRate;
108
109 if ((baud == 0) || ((setup->dataWidth == UART_DATA_9_BIT) && (setup->parity != UART_PARITY_NONE)))
110 {
111 return status;
112 }
113
114 if(srcClkFreq == 0)
115 {
116 srcClkFreq = UART1_FrequencyGet();
117 }
118
119 /* Calculate BRG value */
120 if (brgh == 0)
121 {
122 uxbrg = (((srcClkFreq >> 4) + (baud >> 1)) / baud ) - 1;
123 }
124 else
125 {
126 uxbrg = (((srcClkFreq >> 2) + (baud >> 1)) / baud ) - 1;
127 }
128
129 /* Check if the baud value can be set with low baud settings */
130 if((uxbrg < 0) || (uxbrg > UINT16_MAX))
131 {
132 return status;
133 }
134
135 /* Turn OFF UART1. Save UTXEN, URXEN and UTXBRK bits as these are cleared upon disabling UART */
136
137 status_ctrl = U1STA & (_U1STA_UTXEN_MASK | _U1STA_URXEN_MASK | _U1STA_UTXBRK_MASK);
138
139 U1MODECLR = _U1MODE_ON_MASK;
140
141 if(setup->dataWidth == UART_DATA_9_BIT)
142 {
143 /* Configure UART1 mode */
144 U1MODE = (U1MODE & (~_U1MODE_PDSEL_MASK)) | setup->dataWidth;
145 }
146 else
147 {
148 /* Configure UART1 mode */
149 U1MODE = (U1MODE & (~_U1MODE_PDSEL_MASK)) | setup->parity;
150 }
151
152 /* Configure UART1 mode */
153 U1MODE = (U1MODE & (~_U1MODE_STSEL_MASK)) | setup->stopBits;
154
155 /* Configure UART1 Baud Rate */
156 U1BRG = uxbrg;
157
158 U1MODESET = _U1MODE_ON_MASK;
159
160 /* Re-enable UTXEN, URXEN and UTXBRK. */
161 U1STASET = status_ctrl;
162
163 status = true;
164 }
165
166 return status;
167}

188bool UART1_Read(void* buffer, const size_t size )
189{
190 bool status = false;
191 uint8_t* lBuffer = (uint8_t* )buffer;
192 uint32_t errorStatus = 0;
193 size_t processedSize = 0;
194
195 if(lBuffer != NULL)
196 {
197 /* Clear error flags and flush out error data that may have been received when no active request was pending */
198 UART1_ErrorClear();
199
200 while( size > processedSize )
201 {
202 while(!(U1STA & _U1STA_URXDA_MASK));
203
204 /* Error status */
205 errorStatus = (U1STA & (_U1STA_OERR_MASK | _U1STA_FERR_MASK | _U1STA_PERR_MASK));
206
207 if(errorStatus != 0)
208 {
209 break;
210 }
211 if (( U1MODE & (_U1MODE_PDSEL0_MASK | _U1MODE_PDSEL1_MASK)) == (_U1MODE_PDSEL0_MASK | _U1MODE_PDSEL1_MASK))
212 {
213 /* 9-bit mode */
214 *(uint16_t*)lBuffer = (U1RXREG );
215 lBuffer += 2;
216 }
217 else
218 {
219 /* 8-bit mode */
220 *lBuffer++ = (U1RXREG );
221 }
222
223 processedSize++;
224 }
225
226 if(size == processedSize)
227 {
228 status = true;
229 }
230 }
231
232 return status;
233}

235bool UART1_Write( void* buffer, const size_t size )
236{
237 bool status = false;
238 uint8_t* lBuffer = (uint8_t*)buffer;
239 size_t processedSize = 0;
240
241 if(lBuffer != NULL)
242 {
243 while( size > processedSize )
244 {
245 /* Wait while TX buffer is full */
246 while (U1STA & _U1STA_UTXBF_MASK);
247
248 if (( U1MODE & (_U1MODE_PDSEL0_MASK | _U1MODE_PDSEL1_MASK)) == (_U1MODE_PDSEL0_MASK | _U1MODE_PDSEL1_MASK))
249 {
250 /* 9-bit mode */
251 U1TXREG = *(uint16_t*)lBuffer;
252 lBuffer += 2;
253 }
254 else
255 {
256 /* 8-bit mode */
257 U1TXREG = *lBuffer++;
258 }
259
260 processedSize++;
261 }
262
263 status = true;
264 }
265
266 return status;
267}

269UART_ERROR UART1_ErrorGet( void )
270{
271 UART_ERROR errors = UART_ERROR_NONE;
272
273 errors = (UART_ERROR)(U1STA & (_U1STA_OERR_MASK | _U1STA_FERR_MASK | _U1STA_PERR_MASK));
274
275 if(errors != UART_ERROR_NONE)
276 {
277 UART1_ErrorClear();
278 }
279
280 /* All errors are cleared, but send the previous error state */
281 return errors;
